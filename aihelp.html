            // Function to handle direct, pre-programmed answers
            function handleDirectAnswers(prompt) {
                const lowerCasePrompt = prompt.toLowerCase();

                // Specific website navigation/feature questions
                if (lowerCasePrompt.includes("where do i get books") || lowerCasePrompt.includes("borrow book") || lowerCasePrompt.includes("loan book")) {
                    return "To borrow a book, please click the 'Borrow Book' button in the navigation bar or the 'Borrow Book' link on the homepage. You'll need to fill out a form.";
                } else if (lowerCasePrompt.includes("how do i request help from a human") || lowerCasePrompt.includes("contact help") || lowerCasePrompt.includes("support")) {
                    return "For direct help or to contact a human, please navigate to the 'Help' page in the navigation bar or visit our 'Help Forum'.";
                } else if (lowerCasePrompt.includes("return book") || lowerCasePrompt.includes("give back book")) {
                    return "To return a book, please visit the 'Book Return' page, accessible from the navigation bar.";
                } else if (lowerCasePrompt.includes("search for books") || lowerCasePrompt.includes("find books")) {
                    return "To search for books, please use the 'Search' feature in the navigation bar. You can also explore books by 'Genres'.";
                } else if (lowerCasePrompt.includes("admin login") || lowerCasePrompt.includes("log in as admin")) {
                    return "The 'Admin Login' option is available in the navigation bar for authorized personnel.";
                } else if (lowerCasePrompt.includes("update log") || lowerCasePrompt.includes("what's new")) {
                    return "You can find the latest updates and changes by clicking on the 'Update Log' link in the top navigation bar.";
                } else if (lowerCasePrompt.includes("genres") || lowerCasePrompt.includes("types of books")) {
                    return "Our book collection is categorized by genres such as Fiction, Science Fiction, Mystery, Fantasy, Informational Text, and Biography. You can explore these via the 'Genres' dropdown in the navigation.";
                } else if (lowerCasePrompt.includes("borrowing period") || lowerCasePrompt.includes("how long can i keep a book") || lowerCasePrompt.includes("borrowing guidelines")) {
                    return "Borrowing periods vary: Students can borrow for 2 weeks with one renewal, while Faculty can borrow for as long as needed with renewals, but are requested to keep books in stock. Books are typically dropped off on Wednesdays.";
                }
                // Handle "in stock" questions specifically
                else if (lowerCasePrompt.includes("in stock") || lowerCasePrompt.includes("available") || lowerCasePrompt.includes("do you have") || lowerCasePrompt.includes("is book")) {
                    return "I cannot directly check the live stock or availability of a specific book. Please visit the 'Search' page (library.html) to look for books, or contact the library directly for real-time availability.";
                }
                return null; // No direct answer found
            }
